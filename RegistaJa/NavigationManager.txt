package pt.ulusofona.deisi.cm2223.g21906899_21702200

import android.os.Bundle
import androidx.fragment.app.Fragment
import androidx.fragment.app.FragmentManager
import com.example.projectcm.R


object NavigationManager {

    private fun placeFragment(fm: FragmentManager, fragment: Fragment) {
        val transition = fm.beginTransaction()
        transition.replace(R.id.fragment_container, fragment)
        transition.addToBackStack(null)
        transition.commit()
    }

    fun goToCinemaDetails(fm: FragmentManager, cinemaName: String){
        val fragment = CinemaDetailsFragment().apply {
            arguments = Bundle().apply {
                putString("cinemaName", cinemaName)
            }
        }
        placeFragment(fm, fragment)
    }
    fun goToSearch(fm: FragmentManager){
        placeFragment(fm, PickMovieFragment())
    }

    fun goToSearch2(fm: FragmentManager, value: String) {
        val fragment = PickMovieFragment()
        val bundle = Bundle()
        bundle.putString("key", value)
        fragment.arguments = bundle
        placeFragment(fm, fragment)
    }

    fun goToDetails(fm: FragmentManager, filmeName: String){
        val fragment = DetailsFragment().apply {
            arguments = Bundle().apply {
                putString("movieName", filmeName)
            }
        }
        placeFragment(fm, fragment)
    }
}
